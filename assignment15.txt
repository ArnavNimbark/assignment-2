import random

def main():
    print("=== NUMBER GUESSING GAME ===")
    
    # Initialize best_score to infinity so the first win will always be better
    best_score = float('inf') 

    while True:
        # --- 1. Setup New Game ---
        secret_number = random.randint(1, 100)
        attempts_allowed = 7
        game_won = False
        
        print("\nI have picked a number between 1 and 100.")
        print(f"You have {attempts_allowed} attempts to guess it.")
        if best_score != float('inf'):
            print(f"Current Best Score to beat: {best_score} attempts")

        # --- 2. Guessing Loop ---
        for attempt in range(1, attempts_allowed + 1):
            attempts_remaining = attempts_allowed - attempt
            
            try:
                guess = int(input(f"\nAttempt {attempt}: Enter your guess: "))
            except ValueError:
                print("Please enter a valid number.")
                continue

            # --- Check Condition: WIN ---
            if guess == secret_number:
                print(f"🎉 CONGRATULATIONS! You guessed it in {attempt} attempts.")
                
                # Bonus: Track Best Score
                if attempt < best_score:
                    best_score = attempt
                    print("🏆 New Best Score Record!")
                
                game_won = True
                break # Exit the guessing loop

            # --- Check Condition: HINT (Bonus) ---
            # Check if within 5 numbers (using absolute difference)
            if abs(secret_number - guess) <= 5:
                print("   (HINT: You are very close! Within 5 numbers!)")

            # --- Check Condition: HIGH / LOW ---
            if guess < secret_number:
                print("-> Too Low!")
            else:
                print("-> Too High!")
            
            print(f"   Attempts remaining: {attempts_remaining}")

        # --- 3. End of Game Logic ---
        if not game_won:
            print(f"\n❌ Game Over! You ran out of attempts.")
            print(f"The number was: {secret_number}")

        # --- 4. Play Again? ---
        play_again = input("\nDo you want to play again? (yes/no): ").lower()
        if play_again not in ['yes', 'y']:
            print("Thanks for playing! Goodbye.")
            break

if __name__ == "__main__":
    main()
