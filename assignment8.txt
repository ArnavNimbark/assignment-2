# Ask user for the year
year = int(input("Enter a year to check: "))

# --- Logic Implementation ---
# 1. Check if divisible by 4
if year % 4 == 0:
    # 2. If divisible by 4, check if divisible by 100 (Century year)
    if year % 100 == 0:
        # 3. If divisible by 100, must also be divisible by 400 to be a Leap Year
        if year % 400 == 0:
            print(f"{year}: Leap year")
            print("Reason: Divisible by 400.")
        else:
            print(f"{year}: NOT a leap year")
            print("Reason: Divisible by 100 but NOT divisible by 400.")
    else:
        # Divisible by 4 but not 100 is a standard Leap Year
        print(f"{year}: Leap year")
        print("Reason: Divisible by 4 and NOT divisible by 100.")
else:
    # Not divisible by 4 is never a Leap Year
    print(f"{year}: NOT a leap year")
    print("Reason: Not divisible by 4.")
